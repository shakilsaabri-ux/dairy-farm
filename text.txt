  <script>
    /* ════════════════════════════════════════════════════════
       DATABASE & STATE PERSISTENCE LAYER
    ════════════════════════════════════════════════════════ */
    const STORAGE_KEY = 'DAIRY_FARM_DB_V3';
    let DB = { khila: { S1: {}, S2: {}, "S1-PL": [], "S1-PR": [], "S2-PL": [], "S2-PR": [] } };
    let currentShed = 'S1';
    let curTab = 'page-dash';

    function load() {
      const stored = localStorage.getItem(STORAGE_KEY);
      if (stored) {
        try { 
          DB = JSON.parse(stored); 
          if(!DB.khila) DB.khila = {};
          if(!DB.khila.S1) DB.khila.S1 = {};
          if(!DB.khila.S2) DB.khila.S2 = {};
          ['S1-PL','S1-PR','S2-PL','S2-PR'].forEach(k => { if(!DB.khila[k]) DB.khila[k] = []; });
        } catch(e) { 
          console.error("DB Parsing Error, resetting schema", e); 
        }
      }
    }
    function save() { localStorage.setItem(STORAGE_KEY, JSON.stringify(DB)); }

    /* ════════════════════════════════════════════════════════
       UI SYSTEM NOTIFICATIONS
    ════════════════════════════════════════════════════════ */
    function toast(msg, type = 'success') {
      const t = document.getElementById('toast');
      t.innerText = msg;
      t.className = type === 'err' ? 'show err' : 'show';
      clearTimeout(t.timer);
      t.timer = setTimeout(() => t.classList.remove('show'), 2500);
    }

    /* ════════════════════════════════════════════════════════
       TAB ROUTER ENGINE
    ════════════════════════════════════════════════════════ */
    window.renderPage = (tabId) => {
      curTab = tabId;
      document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
      document.querySelectorAll('.nav-item').forEach(n => {
        n.classList.remove('active');
        if (n.getAttribute('onclick') && n.getAttribute('onclick').includes(tabId)) {
          n.classList.add('active');
        }
      });
      const activePage = document.getElementById(tabId);
      if (activePage) activePage.classList.add('active');

      if (tabId === 'page-dash') renderDash();
      if (tabId === 'page-khila') renderKhila(activePage);
      if (tabId === 'page-more') renderMore(activePage);
    };

    /* ════════════════════════════════════════════════════════
       DASHBOARD CONTROLLER
    ════════════════════════════════════════════════════════ */
    function renderDash() {
      const s1Count = Object.keys(DB.khila.S1).filter(k => DB.khila.S1[k]).length;
      const s2Count = Object.keys(DB.khila.S2).filter(k => DB.khila.S2[k]).length;
      const palaatCount = ['S1-PL', 'S1-PR', 'S2-PL', 'S2-PR'].reduce((acc, curr) => acc + (DB.khila[curr]?.length || 0), 0);

      document.getElementById('ds-s1').innerText = s1Count;
      document.getElementById('ds-s2').innerText = s2Count;
      document.getElementById('ds-palaat').innerText = palaatCount;
      document.getElementById('ds-total').innerText = s1Count + s2Count;
    }

    /* ════════════════════════════════════════════════════════
       KHILA ENGINE CONTROLLER
    ════════════════════════════════════════════════════════ */
    window.switchShed = (shedId) => {
      currentShed = shedId;
      document.getElementById('btn-shed1').classList.toggle('active', shedId === 'S1');
      document.getElementById('btn-shed2').classList.toggle('active', shedId === 'S2');
      
      // Reset filter bars on context switch
      const searchInp = document.getElementById('sf-search');
      const filterSel = document.getElementById('sf-filter');
      if (searchInp) searchInp.value = '';
      if (filterSel) filterSel.value = 'all';

      renderKhila(document.getElementById('page-khila'));
    };

    function renderKhila(container) {
      document.getElementById('khila-grid-title').innerText = `Shed ${currentShed === 'S1' ? '1' : '2'} Layout Matrix`;
      
      document.getElementById('btn-assign-run').onclick = () => assignKhila(currentShed);
      document.getElementById('btn-pl-add').onclick = () => addP(currentShed, 'L');
      document.getElementById('btn-pr-add').onclick = () => addP(currentShed, 'R');

      const matrixBox = document.getElementById('khila-matrix-box');
      matrixBox.innerHTML = '';

      for (let i = 1; i <= 98; i += 10) {
        const row = document.createElement('div');
        row.className = 'khila-row';
        
        let end = Math.min(i + 9, 98);
        for (let j = i; j <= end; j++) {
          const idVal = DB.khila[currentShed][j] || '';
          const cell = document.createElement('div');
          cell.className = `khila-cell ${idVal ? 'occupied' : ''}`;
          cell.innerHTML = `<span class="num">${j}</span><span class="id">${idVal || '-'}</span>`;
          cell.onclick = () => openSheet(j, idVal, currentShed);
          row.appendChild(cell);
        }
        matrixBox.appendChild(row);
      }

      const pBox = document.getElementById('palaat-display-box');
      pBox.innerHTML = `
        <div class="palaat-col">
          <div class="palaat-title">Left Side</div>
          <div class="palaat-list" id="pl-list"></div>
        </div>
        <div class="palaat-col">
          <div class="palaat-title">Right Side</div>
          <div class="palaat-list" id="pr-list"></div>
        </div>
      `;
      renderPalaatSide(currentShed, 'L', 'pl-list');
      renderPalaatSide(currentShed, 'R', 'pr-list');
    }

    function renderPalaatSide(shed, side, listId) {
      const listEl = document.getElementById(listId);
      const data = DB.khila[`${shed}-P${side}`] || [];
      listEl.innerHTML = data.length === 0 ? '<div style="font-size:0.75rem;color:var(--text-muted);text-align:center;padding:8px;">Empty</div>' : '';
      
      data.forEach((item, index) => {
        const row = document.createElement('div');
        row.className = 'palaat-item';
        row.innerHTML = `
          <div style="display:flex; flex-direction:column;">
            <span style="font-weight:600; color:var(--accent);">${item.bufId}</span>
            ${item.note ? `<span style="font-size:0.65rem; color:var(--text-muted);">${item.note}</span>` : ''}
          </div>
          <button class="del-btn" onclick="removeP('${shed}', '${side}', ${index})">&times;</button>
        `;
        listEl.appendChild(row);
      });
    }

    /* ════════════════════════════════════════════════════════
       SEARCH & FILTER MATRIX ENGINE
    ════════════════════════════════════════════════════════ */
    window.filterKhilaMatrix = () => {
      const query = document.getElementById('sf-search').value.trim();
      const filterMode = document.getElementById('sf-filter').value;
      const cells = document.querySelectorAll('.khila-cell');

      cells.forEach(cell => {
        const idSpan = cell.querySelector('.id');
        const bufId = idSpan ? idSpan.innerText.trim() : '-';
        const isOccupied = cell.classList.contains('occupied');

        let matchSearch = true;
        let matchFilter = true;

        if (query !== '' && !bufId.includes(query)) {
          matchSearch = false;
        }

        if (filterMode === 'occupied' && !isOccupied) {
          matchFilter = false;
        } else if (filterMode === 'empty' && isOccupied) {
          matchFilter = false;
        }

        if (matchSearch && matchFilter) {
          cell.style.opacity = '1';
          cell.style.pointerEvents = 'auto';
          if (query !== '' && bufId.includes(query)) {
            cell.style.borderColor = 'var(--success)';
          } else if (isOccupied) {
            cell.style.borderColor = 'rgba(56, 189, 248, 0.4)';
          } else {
            cell.style.borderColor = 'var(--border)';
          }
        } else {
          cell.style.opacity = '0.15';
          cell.style.pointerEvents = 'none';
          cell.style.borderColor = 'var(--border)';
        }
      });
    };

    /* ════════════════════════════════════════════════════════
       MODAL ACTIONS ENGINE & HELPERS
    ════════════════════════════════════════════════════════ */
    window.openSheet = (khilaNum, currentId, shed) => {
      document.getElementById('sh-title').innerText = `Shed ${shed === 'S1' ? '1' : '2'} - Khila #${khilaNum}`;
      const input = document.getElementById('sh-buf-id');
      input.value = currentId;
      
      document.getElementById('sh-btn-save').onclick = () => saveKhilaCell(khilaNum, shed);
      document.getElementById('sh-btn-clear').onclick = () => clearKhila(khilaNum, shed);
      document.getElementById('cell-sheet').classList.add('active');
      input.focus();
    };

    window.closeSheet = () => {
      document.getElementById('cell-sheet').classList.remove('active');
    };

    window.saveKhilaCell = (k, shed) => {
      const val = document.getElementById('sh-buf-id').value.trim();
      if (!DB.khila[shed]) DB.khila[shed] = {};
      
      if (val) DB.khila[shed][k] = val;
      else delete DB.khila[shed][k];
      
      save(); 
      closeSheet(); 
      toast('✅ Saved'); 
      renderKhila(document.getElementById('page-khila')); 
    };

    window.clearKhila = (k, shed) => {
      if (!confirm('Clear this Khila?')) return;
      delete DB.khila[shed][k];
      save(); 
      closeSheet(); 
      toast('🗑 Cleared');
      renderKhila(document.getElementById('page-khila'));
    };

    window.assignKhila = (shed) => {
      const bufId = document.getElementById('ka-buf').value.trim();
      const khila = parseInt(document.getElementById('ka-kh').value);
      if (!bufId || !khila || khila < 1 || khila > 98) {
        toast('Enter valid ID & Khila (1-98)', 'err');
        return;
      }
      if (!DB.khila[shed]) DB.khila[shed] = {};
      DB.khila[shed][khila] = bufId;
      save();
      toast('✅ Assigned');
      document.getElementById('ka-buf').value = '';
      document.getElementById('ka-kh').value = '';
      renderKhila(document.getElementById('page-khila'));
    };

    window.addP = (shed, side) => {
      const inputId = side === 'L' ? 'pl-id' : 'pr-id';
      const inputNote = side === 'L' ? 'pl-note' : 'pr-note';
      const bufId = document.getElementById(inputId).value.trim();
      const note = document.getElementById(inputNote).value.trim();
      
      if (!bufId) { toast('Enter Buffalo ID', 'err'); return; }
      
      const key = `${shed}-P${side}`;
      if (!DB.khila[key]) DB.khila[key] = [];
      DB.khila[key].push({ bufId, note });
      
      save();
      toast('✅ Added to Palaat');
      document.getElementById(inputId).value = '';
      document.getElementById(inputNote).value = '';
      renderKhila(document.getElementById('page-khila'));
    };
